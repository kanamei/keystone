    def initialize(mod, names, zero= nil)
      @module= mod
      @flag_names= names.to_a()
      @zero_name= zero
      for i in 0...@flag_names.size
        mod.const_set(@flag_names[i], Flags.new(1 << i, self))
      end
      mod.const_set(@zero_name, Flags.new(0, self)) if @zero_name
    end
